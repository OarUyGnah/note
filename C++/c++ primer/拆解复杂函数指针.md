## 拆解复杂函数指针

```c
// pfunc是一个函数指针指向的函数参数、返回值都是int
typedef int(*pfunc)(int i);

char* (*(*a)(void))[5];
/*
typedef char*(*p)[5]; // p是一个char*数组的指针
typedef p (*a)(void); // a是一个函数指针，参数void，返回值为p，即一个char*数组的指针
*/

int (*(*v)[])(int);
/*
typedef int(*p)(int); // p是一个函数指针
typedef p (*v)[];	  // v是一个p数组的指针，即函数指针的数组
*/

double (*a(char *(*b)(int,float *), int (*c(int,char *))(int))) (int);
/*
typedef double (*p)(int);
typedef char *(*b)(int,float *); // b是函数指针返回char*，参数是int,float*
typedef int (*c(int,char *))(int); // c是一个函数，c(int,char *)的返回值是函数指针指向一个参数返回值都是int的函数
typedef double (*a(char*,int))(int); // a是一个函数，a(char*,int)的返回值是函数指针，指向函数参数int，返回值double
*/
```
